# ============================
# DATA PREPROCESSING
# ============================

# 1. This removes duplicate rows.
books_df = books_df.drop_duplicates()
kindle_df = kindle_df.drop_duplicates()

# 2. This removes rows with missing essential values.
books_df = books_df.dropna(subset = ["asin", "rating"])
kindle_df = kindle_df.dropna(subset = ["asin", "rating"])

# 3. This filters out invalid ratings. This is due to Amazon ratings needing to be 1-5.
books_df = books_df[(books_df["rating"] >= 1) & (books_df["rating"] <= 5)]
kindle_df = kindle_df[(kindle_df["rating"] >= 1) & (kindle_df["rating"] <= 5)]

# 4. This extracts year and month from reviewTime for time‑based analysis.
books_df["year"] = books_df["reviewTime"].dt.year
kindle_df["year"] = kindle_df["reviewTime"].dt.year

books_df["month"] = books_df["reviewTime"].dt.to_period("M")
kindle_df["month"] = kindle_df["reviewTime"].dt.to_period("M")

# ============================
# BLENDING THE DATASETS
# ============================

# This adds a source label to each dataset.
books_df["source"] = "Books"
kindle_df["source"] = "Kindle"

# This combines into a single dataset.
combined_df = pd.concat([books_df, kindle_df], ignore_index = True)

# This is the preview of the combined dataset for verification.

# Creates a year column.
combined_df["year"] = combined_df["reviewTime"].dt.year

# Computes the yearly average ratings.
yearly_trends = (
    combined_df.groupby(["year", "source"])["rating"]
    .mean()
    .reset_index()
)

plt.figure(figsize = (12, 6))
sns.lineplot(
    data = yearly_trends,
    x = "year",
    y = "rating",
    hue = "source",
    marker = "o",
    palette = "colorblind"
)

plt.title("Average Rating Trend Over Time (Books vs. Kindle)")
plt.xlabel("Year")
plt.ylabel("Average Rating")
plt.tight_layout()

# Saves the figure.
plt.savefig("data_output/yearly_rating_trend.png", dpi = 300)

plt.show()


# ============================
# ADDITIONAL INSIGHT VISUALIZATION
# ============================

# Computes the yearly review counts.
yearly_counts = (
    combined_df.groupby(["year", "source"])["rating"]
    .count()
    .reset_index()
    .rename(columns = {"rating": "review_count"})
)

# This is a pivot for a stacked bar chart.
yearly_pivot = yearly_counts.pivot(
    index = "year",
    columns = "source",
    values = "review_count"
).fillna(0)

plt.figure(figsize = (12, 6))
yearly_pivot.plot(
    kind = "bar",
    stacked = True,
    color = sns.color_palette("colorblind"),
    figsize = (12, 6)
)

plt.title("Review Volume Over Time (Books vs. Kindle)")
plt.xlabel("Year")
plt.ylabel("Number of Reviews")
