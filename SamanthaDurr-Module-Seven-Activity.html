    "UZA SQ Miles",
    "UZA Population",
    "Service Area Population",
    "Service Area SQ Miles",
    "Passenger Miles FY",
    "Unlinked Passenger Trips FY",
    "Avg Trip Length FY",

    "Operating Expenses FY",
    "Avg Cost Per Trip FY",
    "Avg Fares Per Trip FY"
]

# This converts the numeric columns, removes commas, strips spaces and converts to float.
for col in numeric_cols:
    
    if col in filtered_df.columns:
        
        filtered_df[col] = (filtered_df[col]
            .astype(str)
            .str.replace(",", "", regex=False)
            .str.strip())
        
        filtered_df[col] = pd.to_numeric(filtered_df[col], errors = "coerce")

filtered_df_sorted = filtered_df.sort_values("Last Closed FY End Year")

#Creates the line plot.

#Plots the ridership over time for each agency.

            x = "Last Closed FY End Year",
            y = "Unlinked Passenger Trips FY",

# Groups by the agency and sums up the annual ridership.
ridership_by_agency = (filtered_df
    .groupby("Agency", as_index = False)["Unlinked Passenger Trips FY"]
    .sum())

# Creates the bar plot.
plt.figure(figsize = (8, 6))

sns.barplot(data = ridership_by_agency,
    x = "Agency",
    y = "Unlinked Passenger Trips FY",
